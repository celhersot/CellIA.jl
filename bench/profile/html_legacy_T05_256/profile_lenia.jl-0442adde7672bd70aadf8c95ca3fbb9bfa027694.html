69 function read_seed(path)
70 lines = readlines(path)
71 ph, pw = parse.(Int, split(lines[1]))
72 seed = Matrix{Float64}(undef, ph, pw)
73 for i in 1:ph
74 seed[i, :] = parse.(Float64, split(lines[i+1]))
75 end
76 return seed
77 end

79 function tiled_field(dims::Int, seed::Matrix{Float64})
80 A = zeros(Float64, dims, dims)
81 ph, pw = size(seed)
82 if dims < PITCH
83 ox = (dims - ph) ÷ 2; oy = (dims - pw) ÷ 2
84 @inbounds A[ox+1:ox+ph, oy+1:oy+pw] .= seed
85 return A
86 end
87 nt = dims ÷ PITCH
88 io = (PITCH - ph) ÷ 2; jo = (PITCH - pw) ÷ 2
89 for ti in 0:nt-1, tj in 0:nt-1
90 ox = ti * PITCH + io; oy = tj * PITCH + jo
91 @inbounds A[ox+1:ox+ph, oy+1:oy+pw] .= seed
92 end
93 return A
94 end

96 function apply_field!(model, A)
97 for agent in allagents(model)
98 agent.state = A[agent.pos[1], agent.pos[2]]
99 end
100 return model
101 end
