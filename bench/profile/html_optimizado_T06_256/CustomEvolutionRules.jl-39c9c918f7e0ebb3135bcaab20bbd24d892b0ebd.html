118 # necesita el update: el rfft del kernel, los planes FFTW reales (MEASURE) y los buffers
119 # (A, Â, U). Así no se replanifica ni se reasigna memoria en cada paso. Lo llama lenia_init!

121 # Se mantiene :kernel_fft (FFT compleja) por compatibilidad con código que lo lea (p.ej. legacy).
122 function _lenia_build_kernel!(model)
123 dims = size(abmspace(model))
124 R = Int(get(abmproperties(model), :kernel_radius, 13))
125 kfn = get(abmproperties(model), :kernel_fn, _lenia_kernel_fn)
126
127 K = zeros(Float64, dims...)
128 for x in 1:dims[1], y in 1:dims[2]
129 dx = min(x - 1, dims[1] - (x - 1))
130 dy = min(y - 1, dims[2] - (y - 1))
131 r = sqrt(Float64(dx^2 + dy^2)) / R
132 K[x, y] = kfn(r)
133 end
134
135 s = sum(K)
136 s > 0.0 && (K ./= s)
137
138 props = abmproperties(model)
139 props[:kernel_fft] = fft(K) # compatibilidad (lectura externa / legacy)
140
141 # --- estado optimizado: FFT real + planes + buffers reutilizables ---
142 Khat = rfft(K) # rfft del kernel (una sola vez)
143 A = zeros(Float64, dims...) # buffer de campo (se rellena cada paso)
144 Ahat = similar(Khat) # buffer espectral
145 U = Matrix{Float64}(undef, dims...) # buffer de convolución
146 pf = plan_rfft(A; flags = FFTW.MEASURE) # r2c (planificar sobrescribe A; se rellena luego)
147 pb = plan_brfft(Ahat, dims[1]; flags = FFTW.MEASURE) # c2r sin normalizar (la 1/N va aparte)
148 props[:_lenia_state] = (A = A, U = U, Ahat = Ahat, Khat = Khat,
149 pf = pf, pb = pb, invN = 1.0 / (dims[1] * dims[2]))
150 return nothing
151 end

228 # Update FFT estándar + un campo de ruido gaussiano A' = A + η, η ~ N(0, σ), aplicado solo

230 # define `sigma_noise` (o vale 0), por lo que se comporta como lenia_model_step!.

234 # noise_start_step primer paso con ruido (= end ⇒ one-shot; muy grande ⇒ sostenido)
